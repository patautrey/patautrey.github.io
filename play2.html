let audioCtx = null; function ensureAudioContext() { if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } } function midiToFreq(midi) { return 440 * Math.pow(2, (midi - 69) / 12); } // Warm, smooth, non‑distorted tone function createPianoVoice(midi, velocity) { ensureAudioContext(); const ctx = audioCtx; const now = ctx.currentTime; // Envelope (smooth, long tail) const duration = 2.8; const attack = 0.01; const decay = 0.4; const sustain = 0.45; const release = 1.2; // ONE OCTAVE UP const shifted = midi + 12; const freq = midiToFreq(shifted); // Gain envelope const gain = ctx.createGain(); gain.gain.setValueAtTime(0, now); gain.gain.linearRampToValueAtTime(velocity * 0.8, now + attack); gain.gain.linearRampToValueAtTime(velocity * sustain, now + attack + decay); gain.gain.setTargetAtTime(0, now + duration, release); // Warm low‑pass filter const lp = ctx.createBiquadFilter(); lp.type = "lowpass"; lp.frequency.setValueAtTime(2400, now); // smoother, warmer lp.Q.setValueAtTime(0.6, now); // Fundamental + soft 2nd harmonic const osc1 = ctx.createOscillator(); const osc2 = ctx.createOscillator(); osc1.type = "sine"; osc2.type = "sine"; osc1.frequency.setValueAtTime(freq, now); osc2.frequency.setValueAtTime(freq * 2, now); // Gains for partials const g1 = ctx.createGain(); const g2 = ctx.createGain(); g1.gain.value = 1.0; g2.gain.value = 0.18; // softer harmonic = less distortion // Slight random detune for warmth (not EP buzz) osc1.detune.value = (Math.random() - 0.5) * 2; osc2.detune.value = (Math.random() - 0.5) * 3; osc1.connect(g1); osc2.connect(g2); g1.connect(lp); g2.connect(lp); lp.connect(gain); gain.connect(ctx.destination); osc1.start(now); osc2.start(now); const stopTime = now + duration + release * 2; osc1.stop(stopTime); osc2.stop(stopTime); } function playNote(midi, velocity = 1.0) { velocity = Math.max(0.05, Math.min(1.0, velocity)); createPianoVoice(midi, velocity); } window.playNote = playNote; function playCChord() { playNote(60); playNote(64); playNote(67); }
