function playChord(midiNotes, duration = 1.8) { if (!midiNotes || !midiNotes.length) return; if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } const now = audioCtx.currentTime; midiNotes.forEach((midi, index) => { const carrierFreq = midiToFreq(midi); // --- FM Operators --- const carrier = audioCtx.createOscillator(); // main tone const modulator = audioCtx.createOscillator(); // modulator (adds EP character) const modGain = audioCtx.createGain(); // modulation depth const amp = audioCtx.createGain(); // final amplitude envelope const filter = audioCtx.createBiquadFilter(); // --- FM Setup --- carrier.type = "sine"; modulator.type = "sine"; // Classic EP ratio: 1:2 carrier.frequency.value = carrierFreq; modulator.frequency.value = carrierFreq * 2; // Modulation depth (controls brightness) modGain.gain.setValueAtTime(0, now); modGain.gain.linearRampToValueAtTime(80, now + 0.03); // soft EP attack modGain.gain.linearRampToValueAtTime(20, now + 0.25); // mellow sustain modGain.gain.linearRampToValueAtTime(0, now + duration + 0.4); // Connect FM: modulator → modGain → carrier.frequency modulator.connect(modGain); modGain.connect(carrier.frequency); // --- Filter for warmth --- filter.type = "lowpass"; filter.frequency.setValueAtTime(2400, now); filter.frequency.linearRampToValueAtTime(1600, now + 0.4); // --- Amplitude Envelope (EP‑like) --- amp.gain.setValueAtTime(0, now); amp.gain.linearRampToValueAtTime(0.28, now + 0.06); // soft EP attack amp.gain.linearRampToValueAtTime(0.22, now + 0.25); // mellow body amp.gain.linearRampToValueAtTime(0.0, now + duration + 0.5); // smooth release // Routing carrier.connect(filter); filter.connect(amp); amp.connect(audioCtx.destination); // Slight stagger to avoid phase collisions const offset = index * 0.012; carrier.start(now + offset); modulator.start(now + offset); carrier.stop(now + duration + 0.6); modulator.stop(now + duration + 0.6); }); }
