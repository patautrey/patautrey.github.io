function playChord(midiNotes, duration = 1.6) { if (!midiNotes || !midiNotes.length) return; if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } const now = audioCtx.currentTime; midiNotes.forEach((midi, index) => { const baseFreq = midiToFreq(midi); // Fundamental + very soft harmonics (warm, not bell-like) const osc1 = audioCtx.createOscillator(); const osc2 = audioCtx.createOscillator(); const osc3 = audioCtx.createOscillator(); osc1.type = "sine"; osc2.type = "sine"; osc3.type = "sine"; osc1.frequency.value = baseFreq; osc2.frequency.value = baseFreq * 2; // 2nd harmonic (very soft) osc3.frequency.value = baseFreq * 3; // 3rd harmonic (barely audible) const gain1 = audioCtx.createGain(); const gain2 = audioCtx.createGain(); const gain3 = audioCtx.createGain(); // Warm, non-bell harmonic balance gain1.gain.value = 0.75; // strong fundamental gain2.gain.value = 0.12; // soft 2nd harmonic gain3.gain.value = 0.03; // very soft 3rd harmonic // Gentle lowpass for warmth const filter = audioCtx.createBiquadFilter(); filter.type = "lowpass"; filter.frequency.value = 1400; // warm, smooth, no brightness // Soft pad-like envelope const env = audioCtx.createGain(); env.gain.setValueAtTime(0, now); env.gain.linearRampToValueAtTime(0.22, now + 0.28); // slow, warm attack env.gain.linearRampToValueAtTime(0.0, now + duration + 0.45); // long, smooth release // Routing osc1.connect(gain1); osc2.connect(gain2); osc3.connect(gain3); gain1.connect(filter); gain2.connect(filter); gain3.connect(filter); filter.connect(env); env.connect(audioCtx.destination); // Slight stagger to avoid phase collisions const offset = index * 0.012; osc1.start(now + offset); osc2.start(now + offset); osc3.start(now + offset); osc1.stop(now + duration + 0.5); osc2.stop(now + duration + 0.5); osc3.stop(now + duration + 0.5); }); }
